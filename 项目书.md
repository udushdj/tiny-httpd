# 轻量级HTTP服务器与CGI引擎项目书

**目标：5天从零实现可演示的epoll+CGI Web服务器**

---

## 1. 项目概述

| 项目 | 说明 |
|------|------|
| **项目名称** | tiny-httpd（轻量级HTTP/1.1服务器与CGI引擎） |
| **技术栈** | C语言、Linux系统调用（epoll、fork、exec、pipe）、HTTP/1.1协议 |
| **开发周期** | 5天，每天产出可运行的中间版本 |
| **运行环境** | Linux (x86_64 / ARM)，需要 GCC、GNU Make |

### 核心功能

- 基于 epoll 的单线程事件驱动并发（支持 1024+ 并发连接）
- HTTP/1.1 请求解析（GET / POST，支持 Content-Length）
- 静态文件服务（MIME 类型映射、路径穿越防护）
- CGI 动态请求（fork + exec、管道通信、环境变量传递）
- 连接超时管理与资源回收（僵尸进程回收）

| 源码 | `src/main.c` 单文件或多文件拆分 |
| 构建 | Makefile（`make` / `make clean` / `make run`） |

## 2. 模块架构与数据流

### 整体架构图

```
┌─────────────────────────────────────────────────────┐
│                    main() 入口                       │
│  1. 解析命令行参数（端口/DOCROOT）                     │
│  2. 初始化监听 socket + epoll                         │
│  3. 进入事件循环                                      │
└─────────────────────────────────────────────────────┘

客户端请求 → 监听 socket → epoll 事件循环
  ├─ 新连接           → accept() → 加入 epoll (EPOLLIN, 非阻塞)
  ├─ 客户端可读       → 读取数据 → HTTP 解析状态机
  │    └─ 完整请求到达 → 路由判断
  │         ├─ 静态文件  → 路径安全检查 → stat → mmap/read
  │         │              → 构造 HTTP 响应 → 注册写事件(EPOLLOUT)
  │         └─ CGI 请求  → pipe → fork
  │                        ├─ 子进程: exec 脚本，输出到管道
  │                        └─ 父进程: 监听管道(EPOLLIN) → 读取CGI输出
  │                              → 解析头/体 → 构造HTTP响应 → 注册写事件
  ├─ 客户端可写       → 非阻塞 write 循环 → 发送完成后关闭连接
  └─ 超时检查         → 清理空闲连接 + kill CGI 进程 + 回收僵尸
```

### 数据流详解

```
[客户端] --TCP--> [listen_fd] --accept--> [conn_fd]
                                                │
                                    ┌───────────▼───────────┐
                                    │    conn_t.rbuf[]      │
                                    │    (读缓冲 8KB)        │
                                    └───────────┬───────────┘
                                                │
                                    ┌───────────▼───────────┐
                                    │   HTTP 解析状态机      │
                                    │   REQ_LINE → HEADERS   │
                                    │   → BODY → DONE        │
                                    └───────────┬───────────┘
                                                │
                              ┌─────────────────┼─────────────────┐
                              ▼                                   ▼
                    ┌─────────────────┐                   ┌─────────────────┐
                    │   静态文件路由    │                   │    CGI 路由      │
                    │   /index.html   │                   │ /cgi-bin/xxx.py │
                    └───────┬─────────┘                   └───────┬─────────┘
                            │                                     │
                    ┌───────▼─────────┐                   ┌───────▼─────────┐
                    │ stat + mmap     │                   │ fork + exec     │
                    │ 构造 HTTP 响应   │                   │ 管道读取CGI输出  │
                    └───────┬─────────┘                   └───────┬─────────┘
                            │                                     │
                            └─────────────────┬───────────────────┘
                                              ▼
                                    ┌─────────────────┐
                                    │  conn_t.wbuf[]  │
                                    │  (写缓冲 8KB)    │
                                    └───────┬─────────┘
                                            │ EPOLLOUT
                                            ▼
                                    [非阻塞 write 循环]
                                            │
                                            ▼
                                    [关闭连接，释放资源]
```

## 3. 核心数据结构

```c
#define MAX_CONN   1024        // 最大并发连接数
#define BUF_SIZE   8192        // 读写缓冲区大小
#define TIMEOUT    30          // 连接超时秒数
#define MAX_EVENTS 1024        // epoll 事件上限
#define DOCROOT    "./www"     // 默认静态文件根目录

// 连接状态枚举
enum {
    STATE_READING,              // 正在读取请求
    STATE_WRITING,              // 正在发送响应
    STATE_CGI                   // 等待CGI进程输出
};

// 解析器状态枚举
enum {
    PARSE_REQ_LINE,             // 解析请求行
    PARSE_HEADERS,              // 解析头部
    PARSE_BODY,                 // 解析请求体
    PARSE_DONE                  // 解析完成
};

// HTTP 方法枚举
enum {
    HTTP_GET,
    HTTP_POST,
    HTTP_UNKNOWN
};

// 连接上下文
typedef struct {
    int         fd;             // 客户端 socket fd
    int         state;          // STATE_READING / STATE_WRITING / STATE_CGI
    char        rbuf[BUF_SIZE]; // 读缓冲区
    size_t      rpos;           // 读缓冲已消费位置
    size_t      rlen;           // 读缓冲有效数据长度
    char        wbuf[BUF_SIZE]; // 写缓冲区
    size_t      wpos;           // 写缓冲已发送位置
    size_t      wlen;           // 写缓冲有效数据长度
    time_t      last_active;    // 最后活跃时间（用于超时）

    struct http_parser  parser;   // 内嵌解析器
    struct http_request request;  // 解析后的请求

    // CGI 相关
    int         cgi_fd;         // CGI 管道读端 fd
    pid_t       cgi_pid;        // CGI 子进程 PID
    char        cgi_output[BUF_SIZE]; // CGI 输出缓冲区
    size_t      cgi_olen;       // CGI 输出长度
} conn_t;

// HTTP 解析器
typedef struct http_parser {
    int         state;          // 当前解析状态
    int         content_len;    // Content-Length 值
    int         body_read;      // 已读取的 body 字节数
    char        line_buf[2048]; // 行解析临时缓冲
    int         line_pos;       // 当前行写入位置
} http_parser;

// 解析后的 HTTP 请求
typedef struct http_request {
    int         method;         // HTTP_GET / HTTP_POST
    char        uri[1024];      // 请求 URI（不含查询参数）
    char        query[1024];    // 查询参数（? 之后的部分）
    char        version[16];    // HTTP 版本
    char        headers[32][2][256]; // 头部键值对，最多 32 个
    int         header_cnt;     // 头部数量
    char       *body;           // POST 请求体（动态分配）
    size_t      body_len;       // 请求体长度
} http_request;
```

## 5. 关键代码框架

### 5.1 epoll 事件循环骨架

```c
int epfd = epoll_create1(0);
struct epoll_event ev, events[MAX_EVENTS];

// 注册监听 fd
ev.events = EPOLLIN;
ev.data.fd = listen_fd;
epoll_ctl(epfd, EPOLL_CTL_ADD, listen_fd, &ev);

while (1) {
    int nfds = epoll_wait(epfd, events, MAX_EVENTS, 1000); // 1s 超时，用于定时任务

    for (int i = 0; i < nfds; i++) {
        int fd = events[i].data.fd;
        uint32_t evs = events[i].events;

        if (fd == listen_fd) {
            handle_accept(epfd, listen_fd);
        } else {
            conn_t *c = &connections[fd];

            if (evs & (EPOLLIN | EPOLLERR | EPOLLHUP)) {
                handle_read(epfd, c);
            }
            if (evs & EPOLLOUT) {
                handle_write(epfd, c);
            }
        }
    }

    check_timeout(epfd);
    reap_zombies();
}
```

### 5.2 HTTP 解析状态机核心

```c
void parse_request(conn_t *c) {
    char *buf = c->rbuf;

    while (c->rpos < c->rlen) {
        char ch = buf[c->rpos++];

        switch (c->parser.state) {
        case PARSE_REQ_LINE:
            if (ch == '\r') continue;
            if (ch == '\n') {
                // 行结束，解析请求行
                parse_req_line(c);
                c->parser.state = PARSE_HEADERS;
                c->parser.line_pos = 0;
            } else if (c->parser.line_pos < sizeof(c->parser.line_buf) - 1) {
                c->parser.line_buf[c->parser.line_pos++] = ch;
            }
            break;

        case PARSE_HEADERS:
            if (ch == '\r') continue;
            if (ch == '\n') {
                if (c->parser.line_pos == 0) {
                    // 空行 → 头部结束
                    if (c->parser.content_len > 0) {
                        c->parser.state = PARSE_BODY;
                        c->parser.body_read = 0;
                    } else {
                        c->parser.state = PARSE_DONE;
                    }
                } else {
                    parse_header_line(c);
                    c->parser.line_pos = 0;
                }
            } else if (c->parser.line_pos < sizeof(c->parser.line_buf) - 1) {
                c->parser.line_buf[c->parser.line_pos++] = ch;
            }
            break;

        case PARSE_BODY: {
            int remaining = c->parser.content_len - c->parser.body_read;
            int available = c->rlen - c->rpos + 1; // 当前字节已计入
            int to_read = remaining < available ? remaining : available;

            memcpy(c->request.body + c->parser.body_read, &ch, to_read);
            c->parser.body_read += to_read;
            c->rpos += to_read - 1; // 补偿循环的 ++

            if (c->parser.body_read >= c->parser.content_len) {
                c->parser.state = PARSE_DONE;
            }
            break;
        }

        case PARSE_DONE:
            // 已完整解析，等待路由处理
            return;
        }
    }
}
```

### 5.3 静态文件响应流程

```c
void serve_file(epfd, conn_t *c) {
    char full_path[1024];
    char resolved[1024];

    // 1. 拼接路径
    snprintf(full_path, sizeof(full_path), "%s%s", DOCROOT, c->request.uri);

    // 2. 解析真实路径（防穿越）
    if (realpath(full_path, resolved) == NULL) {
        send_error(c, 404, "Not Found");
        return;
    }

    // 3. 安全检查
    if (strncmp(resolved, DOCROOT, strlen(DOCROOT)) != 0) {
        send_error(c, 403, "Forbidden");
        return;
    }

    // 4. stat 文件信息
    struct stat st;
    if (stat(resolved, &st) == -1 || !S_ISREG(st.st_mode)) {
        send_error(c, 404, "Not Found");
        return;
    }

    // 5. mmap 映射文件
    int fd = open(resolved, O_RDONLY);
    char *fdata = mmap(NULL, st.st_size, PROT_READ, MAP_PRIVATE, fd, 0);
    close(fd);

    // 6. MIME 类型
    const char *mime = get_mime_type(resolved);

    // 7. 构造 HTTP 响应头
    int hlen = snprintf(c->wbuf, BUF_SIZE,
        "HTTP/1.1 200 OK\r\n"
        "Content-Type: %s\r\n"
        "Content-Length: %ld\r\n"
        "Connection: close\r\n"
        "Server: tiny-httpd/1.0\r\n"
        "\r\n", mime, st.st_size);

    // 8. 追加文件内容到写缓冲
    memcpy(c->wbuf + hlen, fdata, st.st_size);
    c->wlen = hlen + st.st_size;

    munmap(fdata, st.st_size);

    // 9. 注册写事件
    set_epoll_event(epfd, c, EPOLLOUT);
    c->state = STATE_WRITING;
}
```

### 5.4 CGI 执行流程

```c
void exec_cgi(int epfd, conn_t *c) {
    int pipe_out[2];
    pipe(pipe_out);

    pid_t pid = fork();

    if (pid == 0) {
        // === 子进程 ===
        dup2(pipe_out[1], STDOUT_FILENO);
        close(pipe_out[0]);
        close(pipe_out[1]);

        // 设置 CGI 环境变量
        setenv("REQUEST_METHOD",  http_method_str(c->request.method), 1);
        setenv("QUERY_STRING",    c->request.query, 1);
        setenv("SCRIPT_NAME",     c->request.uri, 1);
        setenv("CONTENT_LENGTH",  "0", 1);
        setenv("SERVER_PROTOCOL", "HTTP/1.1", 1);

        // 构建脚本路径
        char script_path[1024];
        snprintf(script_path, sizeof(script_path), "%s%s", DOCROOT, c->request.uri);

        // 执行 CGI 脚本
        char *argv[] = { script_path, NULL };
        extern char **environ;
        execve(script_path, argv, environ);

        // execve 失败 → 输出错误
        printf("Content-Type: text/plain\r\n\r\nCGI exec failed\n");
        exit(1);
    }

    // === 父进程 ===
    close(pipe_out[1]);

    // 设管道读端为非阻塞
    fcntl(pipe_out[0], F_SETFL, O_NONBLOCK);

    c->cgi_fd = pipe_out[0];
    c->cgi_pid = pid;
    c->state = STATE_CGI;
    c->cgi_olen = 0;

    // 注册管道到 epoll
    struct epoll_event ev;
    ev.events = EPOLLIN;
    ev.data.fd = c->cgi_fd;
    epoll_ctl(epfd, EPOLL_CTL_ADD, c->cgi_fd, &ev);
}
```

```c
void handle_cgi_read(int epfd, conn_t *c) {
    int n = read(c->cgi_fd,
                 c->cgi_output + c->cgi_olen,
                 BUF_SIZE - c->cgi_olen - 1);

    if (n > 0) {
        c->cgi_olen += n;
        c->cgi_output[c->cgi_olen] = '\0';
    } else if (n == 0) {
        // CGI 输出完毕
        char *body = cgi_parse_output(c->cgi_output);
        build_http_response(c, body);   // 200 OK + content-type + body

        // 从 epoll 移除管道
        epoll_ctl(epfd, EPOLL_CTL_DEL, c->cgi_fd, NULL);
        close(c->cgi_fd);
        c->cgi_fd = -1;

        // 注册客户端写事件
        set_epoll_event(epfd, c, EPOLLOUT);
        c->state = STATE_WRITING;
    }
    // n < 0: EAGAIN → 继续等待
}
```

```
Day 1: 网络骨架（epoll + 非阻塞IO）     → 回显服务器可演示
Day 2: HTTP解析 + 静态文件              → 可浏览网页
```

**核心原则**：
1. 每日结束都有可运行的版本，风险可控
2. 先跑通主线，再完善边界情况
3. 每次代码变更后立即编译测试
4. 善用 `strace` / `valgrind` / `curl -v` 调试

现在确认项目书内容无误后，即可开始 Day 1 的编码工作。
